import { directusGraphqlWithToken } from '@/lib/directus';


export interface CaseObservableInput {
  observable_type: string;
  observable_value: string;
  description?: string;
  source?: string;
  risk_level?: string;
  confidence?: number;
}

export interface CaseArtifactInput {
  name: string;
  artifact_type: string;
  description?: string;
  source?: string;
  relevance_score?: number;
}

export interface CaseEntityInput {
  entity_name: string;
  entity_type: string;
  description?: string;
  source_system?: string;
  risk_score?: number;
}

export interface CaseCreateGqlInput {
  display_name: string;
  description?: string;
  case_type: string;
  status: string;
  priority: 'low' | 'medium' | 'high' | 'urgent';
  domain?: string;
  owned_by?: string;
  assigned_to?: string;
  due_at?: string;
  risk_score?: number;
  case_number?: string;
  observables: CaseObservableInput[];
  artifacts: CaseArtifactInput[];
  entities: CaseEntityInput[];
}


function normaliseObservable(r: any) {
  return {
    id:           r.id,
    type:         r.observable_type        || 'ip',
    value:        r.observable_value       || '',
    description:  r.description            || '',
    source:       r.attributes?.source     || '',
    riskLevel:    r.attributes?.risk_level || 'medium',
    confidence:   r.attributes?.confidence ?? 0.8,
    tags:         [],
    dateObserved: r.date_created || new Date().toISOString(),
  };
}

function normaliseArtifact(r: any) {
  return {
    id:             r.id,
    name:           r.name          || '',
    type:           r.artifact_type || 'document',
    description:    r.description   || '',
    source:         r.source        || '',
    relevanceScore: r.metadata?.relevance_score ?? 0.8,
    tags:           r.tags          || [],
    dateCreated:    r.date_created  || new Date().toISOString(),
  };
}

function normaliseEntity(r: any) {
  return {
    id:             r.id,
    name:           r.entity_name      || '',
    type:           r.entity_type      || 'person',
    description:    r.description      || '',
    source:         r.source_system    || '',
    riskScore:      r.risk_score       ?? 0,
    confidence:     r.confidence_score ?? 0.8,
    tags:           [],
    dateIdentified: r.date_created     || new Date().toISOString(),
  };
}

export { normaliseObservable, normaliseArtifact, normaliseEntity };


const PRIORITY_INT: Record<string, number>  = { urgent: 1, high: 2, medium: 3, low: 4 };
const PRIORITY_LABEL: Record<number, string> = { 1: 'urgent', 2: 'high', 3: 'medium', 4: 'low' };


function esc(s: string | null | undefined): string {
  if (!s) return '';
  return s
    .replace(/\\/g, '\\\\')
    .replace(/"/g, '\\"')
    .replace(/\n/g, '\\n')
    .replace(/\r/g, '');
}


function obsFragment(o: CaseObservableInput): string {
  return `{
        observables_id: {
          observable_type: "${esc(o.observable_type)}"
          observable_value: "${esc(o.observable_value)}"
          description: "${esc(o.description)}"
          attributes: { source: "${esc(o.source)}", risk_level: "${esc(o.risk_level)}", confidence: ${o.confidence ?? 0.8} }
        }
      }`;
}

function artFragment(a: CaseArtifactInput): string {
  return `{
        artifacts_id: {
          name: "${esc(a.name)}"
          artifact_type: "${esc(a.artifact_type)}"
          description: "${esc(a.description)}"
          source: "${esc(a.source)}"
          metadata: { relevance_score: ${a.relevance_score ?? 0.8} }
        }
      }`;
}

function entFragment(e: CaseEntityInput): string {
  return `{
        entities_id: {
          entity_name: "${esc(e.entity_name)}"
          entity_type: "${esc(e.entity_type)}"
          description: "${esc(e.description)}"
          source_system: "${esc(e.source_system)}"
          risk_score: ${e.risk_score ?? 0}
        }
      }`;
}


export function buildCreateCaseMutation(input: CaseCreateGqlInput): string {
  const priorityInt = PRIORITY_INT[input.priority] ?? 3;

  const optFields: string[] = [];
  if (input.description)              optFields.push(`description: "${esc(input.description)}"`);
  if (input.domain)                   optFields.push(`domain: "${esc(input.domain)}"`);
  if (input.owned_by)                 optFields.push(`owned_by: "${esc(input.owned_by)}"`);
  if (input.assigned_to)              optFields.push(`assigned_to: "${esc(input.assigned_to)}"`);
  if (input.due_at)                   optFields.push(`due_at: "${esc(input.due_at)}"`);
  if (input.risk_score !== undefined) optFields.push(`risk_score: ${input.risk_score}`);
  if (input.case_number)              optFields.push(`case_number: "${esc(input.case_number)}"`);

  // Directus M2M nested create format: { create: [...], update: [], delete: [] }
  if (input.observables.length > 0) {
    optFields.push(`case_observables: { create: [${input.observables.map(obsFragment).join(', ')}], update: [], delete: [] }`);
  }
  if (input.artifacts.length > 0) {
    optFields.push(`case_artifacts: { create: [${input.artifacts.map(artFragment).join(', ')}], update: [], delete: [] }`);
  }
  if (input.entities.length > 0) {
    optFields.push(`case_entities: { create: [${input.entities.map(entFragment).join(', ')}], update: [], delete: [] }`);
  }

  return `mutation {
  create_cases_item(data: {
    display_name: "${esc(input.display_name)}"
    case_type: "${esc(input.case_type)}"
    status: "${esc(input.status)}"
    priority: ${priorityInt}
    ${optFields.join('\n    ')}
  }) {
    id
    display_name
    description
    case_type
    status
    priority
    domain
    owned_by
    assigned_to
    due_at
    risk_score
    case_number
    date_created
    case_observables {
      observables_id {
        id
        observable_type
        observable_value
        description
        attributes
      }
    }
    case_artifacts {
      artifacts_id {
        id
        name
        artifact_type
        description
        source
        metadata
      }
    }
    case_entities {
      entities_id {
        id
        entity_name
        entity_type
        description
        source_system
        risk_score
      }
    }
  }
}`;
}


export async function createCaseGql(input: CaseCreateGqlInput): Promise<any> {
  const client = directusGraphqlWithToken as any;
  if (!client?.query || typeof client.query !== 'function') {
    throw new Error('Directus GraphQL client is not configured');
  }

  const mutation = buildCreateCaseMutation(input);
  console.log('[createCaseGql] executing mutation for:', input.display_name);

  try {
    const result = await client.query(mutation);
    return result;
  } catch (error: any) {
    const gqlErr = error?.errors?.[0];
    if (gqlErr) {
      const detail = gqlErr.extensions?.errors?.[0]?.message ?? gqlErr.message;
      throw new Error(`GraphQL Error: ${detail}`);
    }
    throw new Error('Case creation failed', { cause: error });
  }
}

// ── Query: list cases ─────────────────────────────────────────────────────────

export async function getCasesGql(search?: string): Promise<any[]> {
  const client = directusGraphqlWithToken as any;
  if (!client?.query || typeof client.query !== 'function') {
    throw new Error('Directus GraphQL client is not configured');
  }

  const filterArg = search
    ? `(filter: { _or: [{ display_name: { _icontains: "${esc(search)}" } }, { case_number: { _icontains: "${esc(search)}" } }, { description: { _icontains: "${esc(search)}" } }] }, sort: ["-date_created"])`
    : `(sort: ["-date_created"])`;

  const query = `query {
  cases${filterArg} {
    id
    display_name
    description
    case_type
    status
    priority
    domain
    owned_by
    assigned_to
    due_at
    risk_score
    case_number
    date_created
    date_updated
    case_observables {
      observables_id {
        id
        observable_type
        observable_value
        description
        attributes
      }
    }
    case_artifacts {
      artifacts_id {
        id
        name
        artifact_type
        description
        source
        metadata
        tags
      }
    }
    case_entities {
      entities_id {
        id
        entity_name
        entity_type
        description
        source_system
        risk_score
        confidence_score
      }
    }
  }
}`;

  try {
    const result = await client.query(query);
    const rows: any[] = result?.cases ?? [];

    return rows.map(r => ({
      id:              r.id,
      title:           r.display_name || 'Untitled',
      description:     r.description  || '',
      case_type:       r.case_type    || 'fraud',
      case_id:         r.case_number  || r.id,
      priority:        PRIORITY_LABEL[r.priority] ?? r.priority ?? 'medium',
      status:          r.status       || 'open',
      category:        r.domain       || 'General',
      owner:           r.owned_by     || undefined,
      secondary_owner: r.assigned_to  || undefined,
      sla_due_date:    r.due_at       || undefined,
      risk_score:      r.risk_score   ?? undefined,
      date_created:    r.date_created || undefined,
      date_updated:    r.date_updated || undefined,
      last_activity:   r.date_updated || r.date_created || undefined,
      progress:        10,
      observables: (r.case_observables || []).map((j: any) => j.observables_id).filter(Boolean).map(normaliseObservable),
      artifacts:   (r.case_artifacts   || []).map((j: any) => j.artifacts_id).filter(Boolean).map(normaliseArtifact),
      entities:    (r.case_entities    || []).map((j: any) => j.entities_id).filter(Boolean).map(normaliseEntity),
    }));
  } catch (error: any) {
    const gqlErr = error?.errors?.[0];
    if (gqlErr) {
      const detail = gqlErr.extensions?.errors?.[0]?.message ?? gqlErr.message;
      throw new Error(`GraphQL Error: ${detail}`);
    }
    throw new Error('Failed to fetch cases', { cause: error });
  }
}
